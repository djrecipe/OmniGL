773 GLint Scene::AddLight(GLfloat * light_position, GLfloat * light_color, GLfloat * ambient_color, BlenderObj* renobj_in)
774 {
775  glm::mat4 mvmat=this->matrices.translation_rotation_scale;
776  GLfloat position_out[4]={0.0,0.0,0.0,0.0};
