637  these_indices[i%3]=this->element_indices[i];
638  if((i+1)%3==0)
639  {
640  glm::vec3 vec_u(this->vertices[this->element_indices[i-1]].x-this->vertices[this->element_indices[i-2]].x,this->vertices[this->element_indices[i-1]].y-this->vertices[this->element_indices[i-2]].y,this->vertices[this->element_indices[i-1]].z-this->vertices[this->element_indices[i-2]].z);

642  glm::vec3 vec_norm;
643  vec_norm.x=vec_u.y*vec_v.z-vec_u.z*vec_v.y;
644  vec_norm.y=vec_u.z*vec_v.x-vec_u.x*vec_v.z;

646  if(!(vec_norm.x==0.0 && vec_norm.y==0.0 && vec_norm.z==0.0))
647  vec_norm=glm::normalize(vec_norm);
648  for(int j=0; j<3; j++)
649  this->field[these_indices[j]]->AddNormal(vec_norm);
650  }
651  }
652  //
653  std::vector<glm::vec3>::iterator j;
654  this->normals.clear();
655  for(int i=0; i<this->field.size(); i++)
656  {
657  std::vector<glm::vec3> curr_normals=this->field[i]->normals;

661  {

663  missing_normals++;
664  else
665  blended_normal+=(*j);
666  }
667  blended_normal/=(curr_normals.size()-missing_normals);
668  this->normals.push_back(blended_normal);
669  }
